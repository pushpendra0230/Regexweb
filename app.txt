                <h2 className="text-xl font-semibold mb-2">{item.title}</h2>
                <p>{item.desc}</p>
              </div>
            ))}
          </div>
        </div>
      </header>

      {/* StatsHeader Section */}
      <StatsHeader />

      <EnquiryForm />
